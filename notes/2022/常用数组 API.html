    # 1.push()

    # 向数组的末尾添加一个或多个元素，并返回新数组的长度(改变原数组)

    let arr1 = [1,2,3]
let arr2 = arr1.push(4,5)
console.log(arr1)  //  [1,2,3,4,5]
console.log(arr2)  //  5

    # 2.unshift()

    # 向数组的开头添加一个或多个元素，并返回新数组的长度(改变原数组)

    let arr1 = [1,2,3]
let arr2 = arr1.unshift(4,5)
console.log(arr1)  //  [4,5,1,2,3]
console.log(arr2)  //  5

    # 7.sort()

    # 对数组的元素进行排序,返回排序后的新数组(改变原数组)

    # 默认按照数组元素第一位的 ASCII 码从小到大排列，只限数组中是数字或者数字字符串

    let arr = [1,7,2,6,3,5,4]
let brr = arr.sort()
console.log(brr)   //  [1,2,3,4,5,6,7]
let crr = arr.sort((a,b) => a-b)
console.log(crr)   //  [1,2,3,4,5,6,7]  升序
let drr = arr.sort((a,b) => b-a)
console.log(drr)   //  [7,6,5,4,3,2,1]  降序

    # 9.slice(start,end)

    # 截取数组(不改变原数组)

    # slice(start,end) start:截取数组单元起始位置的索引 end:截取数组单元结束位置的索引

    let arr = [1,2,3,4,5,6]
console.log(arr.slice(1))      //  [2,3,4,5,6],只有一个参数且不小于0时，则从此索引开始截取到数组的末尾
console.log(arr.slice(-1))     //  [6],只有一个参数且小于0时，则从倒数开始位截取到数组的末尾
console.log(arr.slice(1,3))    //  [2,3],两个参数时，从开始下标截取到结束下标，不包含结束下标
console.log(arr.slice(1,-1))   //  [2,3,4,5],从第一位截取到倒数第一位
console.log(arr.slice(-2,-1))  //  [5]  倒数第2位截取到倒数第1位
console.log(arr.slice(-1,1))   //  反向截取，不合法，返回空数组
console.log(arr.slice(-1,-2))  //  反向截取，不合法，返回空数组

    # 10.splice(index,count,val1,val2...)

    # 删除(插入、更新)数组中的元素(改变原数组)

    # splice(index,count,val1,val2...) index:索引 count:删除元素数量,val1,val2...插入的新元素

    let arr = [1,2,3,4,5,6]
console.log(arr.splice(1,2,99)) //  [2,3] 从下标为1 删除2个元素 返回被删除的数组
console.log(arr)                //  [1,99,4,5,6]   原数组1的位置被插入99这个元素

let arr = [1,2,3,4,5,6]
console.log(arr.splice(0))      //  [1,2,3,4,5,6]  当参数为单个且不小于0时，将截取到数组末尾
console.log(arr)                //  []

let arr = [1,2,3,4,5,6]
console.log(arr.splice(-1))      // [6]            当参数为单个且小于0时，将从倒数开始位截取到数组的末尾
console.log(arr)                 // [1,2,3,4,5]

let arr = [1,2,3,4,5,6]
console.log(arr.splice(-1,2))      // [6]         从倒数-1开始位截取2个元素
console.log(arr)                 // [1,2,3,4,5]

    # 12.indexOf(value)

    # 从索引为 0 开始，检查数组中是否包含有 value，有则返回匹配到的第一个索引，没有则返回-1(不改变原数组)

    let arr = [1,2,3,4,3,5]
console.log(arr.indexOf(3))     //  2   数组中有2个3，第一个3索引为2
console.log(arr.indexOf(99))    //  -1  数组中没有99，返回-1

    # 13.lastIndexOf(value)

    # 从最后的索引开始，检查数组中是否包含有 value，有则返回匹配到的第一个索引，没有则返回-1(不改变原数组)

    let arr = [1,2,3,4,3,5]
console.log(arr.lastIndexOf(3))     //  4   从最后的索引反过来查找3，第一个3的下标为4
console.log(arr.lastIndexOf(99))    //  -1  数组中没有99，返回-1

    # 19.reduce(funciton(previousValue,currentValue, index, arr),initialValue)

    # 接收一个函数作为累加器，数组中的每个值（从左到右）开始缩减，最终计算为一个值

    # 如果initialValue省略，则previousValue和currentValue分别为数组中的第一项元素和第二项元素；

    # 如果initialValue存在，则previousValue为initialValue，而currentValue为数组中的第一项

    let arr = [1,2,3,4,5]
let brr = arr.reduce((previousValue,currentValue) =>{
    return previousValue + currentValue
})
console.log(brr)    //  15
let arr = [1,2,3,4,5]
let brr = arr.reduce((previousValue,currentValue) =>{
    return previousValue + currentValue
},100)
console.log(brr)    //  115
